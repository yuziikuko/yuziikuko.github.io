                                                    五、简化对象写法

                                                    郁子小于 1 分钟约 147 字笔记ECMAScript尚硅谷李强
                                                    此页内容
                                                    • （一）属性名和变量名相同时可以省略，只写一次
                                                    • （二）方法的声明方式可以简化
                                                    • ES6 允许在大括号里面直接写入变量和函数，作为对象的属性和方法，更加简洁

                                                    # （一）属性名和变量名相同时可以省略，只写一次

                                                    let name = "ikuko";
let change = function () {
  console.log("最美不过二次元!");
};

const love = {
  // name: name,
  // change: change,
  name,
  change,
};
console.log(love);

                                                    # （二）方法的声明方式可以简化

                                                    let name = "ikuko";
let change = function () {
  console.log("最美不过二次元!");
};

const love = {
  // watchCartoon: function() {
  //   console.log("看日漫。");
  // },
  watchCartoon() {
    console.log("看日漫。");
  },
};
console.log(love);
