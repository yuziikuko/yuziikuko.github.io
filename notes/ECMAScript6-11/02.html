                                                            • （一）声明常量
                                                            • （二）特性
                                                              • 1.一定要赋初始值
                                                              • 2.一般常量使用大写（潜规则）
                                                              • 3.常量值不可修改
                                                              • 4.块级作用域
                                                              • 5.对于数组和对象的元素修改不算做对常量的修改，不会报错

                                                            # （一）声明常量

                                                            const ikuko = "郁子";
console.log(ikuko);

                                                            # （二）特性

                                                            # 1.一定要赋初始值

                                                            const A;
// Missing initializer in const declaration

                                                            # 2.一般常量使用大写（潜规则）

                                                            const a = 100; // 不推荐
const A = 100; // 推荐

                                                            # 3.常量值不可修改

                                                            ikuko = "樱井";
// Assignment to constant variable.

                                                            # 4.块级作用域

                                                            {
  const PLAYER = "UZI";
}
console.log(PLAYER);
// PLAYER is not defined

                                                            # 5.对于数组和对象的元素修改不算做对常量的修改，不会报错

                                                            • 因为当前数组或对象指向的内存地址不发生改变。
                                                            const TEAM = ["UZI", "MXLG", "Ming", "Letme"];
TEAM.push("Meiko"); // ok
console.log(TEAM);

TEAM = 100; // Assignment to constant variable.
